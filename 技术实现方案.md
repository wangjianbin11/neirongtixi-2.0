## 二、技术栈选型

### 2.1 前端技术栈

```yaml
核心框架: React 18.x
语言: TypeScript 5.x
状态管理: Zustand + React Query
UI组件库: Ant Design 5.x
构建工具: Vite 5.x
样式方案: Tailwind CSS + CSS Modules
图表库: ECharts / Recharts
表单处理: React Hook Form + Zod
路由: React Router v6
HTTP客户端: Axios
实时通信: Socket.io-client
```

### 2.2 后端技术栈

```yaml
运行时: Node.js 20 LTS
语言: TypeScript 5.x
Web框架: Express 4.x
API文档: Swagger/OpenAPI
数据验证: Zod
认证授权: JWT + Passport.js
日志: Winston + Morgan
进程管理: PM2
消息队列: Bull (Redis-based)
定时任务: Agenda (MongoDB-based)
```

### 2.3 AI服务技术栈

```yaml
运行时: Python 3.11+
框架: FastAPI
AI/LLM:
  - OpenAI API (GPT-4/GPT-3.5)
  - Anthropic API (Claude)
  - 本地模型 Ollama (可选)
向量数据库: Qdrant / Weaviate (用于RAG)
文本处理: LangChain
异步处理: Celery + Redis
```

### 2.4 数据库技术栈

```yaml
关系型数据库: PostgreSQL 15+
  - 用途: 用户数据、关键词库、话题库、系统配置
  - ORM: Prisma / TypeORM

文档数据库: MongoDB 7+
  - 用途: 内容存储、发布记录、分析数据
  - ODM: Mongoose

缓存: Redis 7+
  - 用途: 会话、队列、热点数据缓存
  - 客户端: ioredis

全文搜索: Elasticsearch 8+
  - 用途: 全文搜索、内容检索、日志分析

文件存储: MinIO / AWS S3
  - 用途: 图片、视频、文档存储
```

### 2.5 开发工具链

```yaml
版本控制: Git + GitHub/GitLab
CI/CD: GitHub Actions / GitLab CI
容器化: Docker + Docker Compose
API测试: Postman / Bruno
监控: Prometheus + Grafana
日志聚合: ELK Stack
错误追踪: Sentry
```

## 三、数据库设计

### 3.1 PostgreSQL 数据模型 (结构化数据)

```sql
-- ============================================
-- 用户与权限模块
-- ============================================

-- 用户表
CREATE TABLE users (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    username VARCHAR(50) UNIQUE NOT NULL,
    email VARCHAR(255) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    role VARCHAR(20) NOT NULL DEFAULT 'user', -- admin, editor, publisher, viewer
    status VARCHAR(20) NOT NULL DEFAULT 'active', -- active, inactive, suspended
    avatar_url VARCHAR(500),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 用户会话表
CREATE TABLE user_sessions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES users(id) ON DELETE CASCADE,
    token VARCHAR(500) UNIQUE NOT NULL,
    expires_at TIMESTAMP NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- ============================================
-- 关键词模块
-- ============================================

-- 关键词表
CREATE TABLE keywords (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    keyword VARCHAR(255) NOT NULL,
    search_volume INTEGER DEFAULT 0,
    kd_score DECIMAL(5,2), -- Keyword Difficulty
    cpc DECIMAL(10,2), -- Cost Per Click
    intent VARCHAR(30) NOT NULL, -- informational, commercial, transactional, navigational
    asg_relevance SMALLINT CHECK (asg_relevance BETWEEN 1 AND 10),
    priority VARCHAR(5) CHECK (priority IN ('S', 'A', 'B', 'C')),
    status VARCHAR(20) DEFAULT 'pending', -- pending, selected, in_use, completed
    source VARCHAR(100), -- ahrefs, semrush, manual
    tags TEXT[], -- PostgreSQL array
    notes TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(keyword)
);

CREATE INDEX idx_keywords_status ON keywords(status);
CREATE INDEX idx_keywords_priority ON keywords(priority);
CREATE INDEX idx_keywords_intent ON keywords(intent);
CREATE INDEX idx_keywords_tags ON keywords USING GIN(tags);

-- 关键词分类表
CREATE TABLE keyword_categories (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name VARCHAR(100) UNIQUE NOT NULL,
    description TEXT,
    parent_id UUID REFERENCES keyword_categories(id),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 关键词-分类关联表
CREATE TABLE keyword_category_relations (
    keyword_id UUID REFERENCES keywords(id) ON DELETE CASCADE,
    category_id UUID REFERENCES keyword_categories(id) ON DELETE CASCADE,
    PRIMARY KEY (keyword_id, category_id)
);

-- ============================================
-- 话题模块
-- ============================================

-- 话题表
CREATE TABLE topics (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    title VARCHAR(500) NOT NULL,
    description TEXT,
    topic_type VARCHAR(30) NOT NULL, -- tutorial, qa, case_study, insight, review, comparison
    target_customer VARCHAR(30), -- startup, experienced, team, local
    priority VARCHAR(5) CHECK (priority IN ('S', 'A', 'B', 'C')),
    status VARCHAR(20) DEFAULT 'pending', -- pending, approved, in_production, completed, published
    estimated_effort INTEGER, -- 分钟
    created_by UUID REFERENCES users(id),
    assigned_to UUID REFERENCES users(id),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    published_at TIMESTAMP
);

CREATE INDEX idx_topics_status ON topics(status);
CREATE INDEX idx_topics_priority ON topics(priority);
CREATE INDEX idx_topics_type ON topics(topic_type);

-- 话题-关键词关联表
CREATE TABLE topic_keywords (
    topic_id UUID REFERENCES topics(id) ON DELETE CASCADE,
    keyword_id UUID REFERENCES keywords(id) ON DELETE CASCADE,
    is_primary BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (topic_id, keyword_id)
);

-- 话题-平台分配表
CREATE TABLE topic_platforms (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    topic_id UUID REFERENCES topics(id) ON DELETE CASCADE,
    platform VARCHAR(30) NOT NULL, -- youtube, tiktok, blog, twitter, linkedin, reddit, quora
    content_format VARCHAR(50), -- long_video, short_video, article, post, answer
    status VARCHAR(20) DEFAULT 'pending',
    scheduled_at TIMESTAMP,
    published_at TIMESTAMP,
    url VARCHAR(500),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(topic_id, platform)
);

-- ============================================
-- 技能模块
-- ============================================

-- 技能模板表
CREATE TABLE skill_templates (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    code VARCHAR(50) UNIQUE NOT NULL, -- Skill-01, Skill-02, etc.
    name VARCHAR(100) NOT NULL,
    description TEXT,
    category VARCHAR(50), -- data_input, content_production, distribution, optimization
    prompt_template TEXT NOT NULL,
    input_schema JSONB, -- 输入参数结构
    output_schema JSONB, -- 输出结构
    version VARCHAR(20) DEFAULT '1.0',
    is_active BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 技能执行记录表
CREATE TABLE skill_executions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    skill_code VARCHAR(50) NOT NULL,
    input_data JSONB,
    output_data JSONB,
    status VARCHAR(20) DEFAULT 'pending', -- pending, processing, completed, failed
    error_message TEXT,
    execution_time_ms INTEGER,
    cost_usd DECIMAL(10,4),
    created_by UUID REFERENCES users(id),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    completed_at TIMESTAMP
);

CREATE INDEX idx_skill_executions_status ON skill_executions(status);
CREATE INDEX idx_skill_executions_skill ON skill_executions(skill_code);

-- ============================================
-- 内容模块
-- ============================================

-- 内容表
CREATE TABLE contents (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    topic_id UUID REFERENCES topics(id) ON DELETE SET NULL,
    title VARCHAR(500) NOT NULL,
    content_type VARCHAR(30) NOT NULL, -- article, video_script, social_post, forum_answer
    platform VARCHAR(30) NOT NULL,
    status VARCHAR(20) DEFAULT 'draft', -- draft, review, approved, published, archived
    content_text TEXT,
    content_metadata JSONB, -- 存储额外的结构化数据
    generated_by_skill VARCHAR(50),
    created_by UUID REFERENCES users(id),
    reviewed_by UUID REFERENCES users(id),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    published_at TIMESTAMP
);

CREATE INDEX idx_contents_status ON contents(status);
CREATE INDEX idx_contents_platform ON contents(platform);
CREATE INDEX idx_contents_topic ON contents(topic_id);

-- 内容版本表
CREATE TABLE content_versions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    content_id UUID REFERENCES contents(id) ON DELETE CASCADE,
    version_number INTEGER NOT NULL,
    content_text TEXT NOT NULL,
    change_summary TEXT,
    created_by UUID REFERENCES users(id),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(content_id, version_number)
);

-- ============================================
-- 素材模块
-- ============================================

-- 素材表
CREATE TABLE assets (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    title VARCHAR(255) NOT NULL,
    type VARCHAR(30) NOT NULL, -- case, data, quote, image, video, template
    category VARCHAR(50),
    description TEXT,
    file_url VARCHAR(500),
    file_size BIGINT,
    file_type VARCHAR(50),
    tags TEXT[],
    source VARCHAR(100),
    is_active BOOLEAN DEFAULT TRUE,
    created_by UUID REFERENCES users(id),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_assets_type ON assets(type);
CREATE INDEX idx_assets_tags ON assets USING GIN(tags);

-- ============================================
-- 发布模块
-- ============================================

-- 发布任务表
CREATE TABLE publish_tasks (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    content_id UUID REFERENCES contents(id) ON DELETE CASCADE,
    platform VARCHAR(30) NOT NULL,
    status VARCHAR(20) DEFAULT 'pending', -- pending, scheduled, publishing, published, failed
    scheduled_at TIMESTAMP,
    published_at TIMESTAMP,
    platform_post_id VARCHAR(255),
    platform_post_url VARCHAR(500),
    error_message TEXT,
    retry_count INTEGER DEFAULT 0,
    created_by UUID REFERENCES users(id),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_publish_tasks_status ON publish_tasks(status);
CREATE INDEX idx_publish_tasks_scheduled ON publish_tasks(scheduled_at);

-- ============================================
-- 数据与分析模块
-- ============================================

-- 平台数据表
CREATE TABLE platform_metrics (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    platform VARCHAR(30) NOT NULL,
    content_id UUID REFERENCES contents(id) ON DELETE SET NULL,
    metric_date DATE NOT NULL,
    views INTEGER DEFAULT 0,
    clicks INTEGER DEFAULT 0,
    likes INTEGER DEFAULT 0,
    comments INTEGER DEFAULT 0,
    shares INTEGER DEFAULT 0,
    saves INTEGER DEFAULT 0,
    engagement_rate DECIMAL(5,2),
    ctr DECIMAL(5,2),
    metadata JSONB,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(platform, content_id, metric_date)
);

CREATE INDEX idx_platform_metrics_date ON platform_metrics(metric_date);
CREATE INDEX idx_platform_metrics_platform ON platform_metrics(platform);

-- 关键词排名数据表
CREATE TABLE keyword_rankings (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    keyword_id UUID REFERENCES keywords(id) ON DELETE CASCADE,
    platform VARCHAR(30) NOT NULL, -- google, youtube, etc.
    position INTEGER,
    search_volume INTEGER,
    url VARCHAR(500),

CREATE INDEX idx_keyword_rankings_keyword ON keyword_rankings(keyword_id);
CREATE INDEX idx_keyword_rankings_checked ON keyword_rankings(checked_at);

-- ============================================
-- 系统配置模块
-- ============================================

-- 系统配置表
CREATE TABLE system_configs (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),

    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- API密钥表
CREATE TABLE api_keys (

-- ============================================
-- 工作流与调度模块
-- ============================================

-- 工作流定义表

CREATE TABLE workflow_executions (

-- ============================================

CREATE TABLE notifications (
```


### 3.3 Redis 数据结构

```redis
# ============================================
# 缓存键命名规范
# ============================================

# 用户会话
session:{user_id}:{session_token} -> Hash (TTL: 24h)
  fields: user_data, created_at, expires_at

# 热点关键词缓存
```

```typescript
// 技能执行流程
class SkillExecutionEngine {
  async executeSkill(skillCode: string, input: any, options?: ExecuteOptions) {
    // 1. 获取技能模板
    const template = await this.getSkillTemplate(skillCode);

    // 2. 验证输入
    this.validateInput(input, template.inputSchema);

    // 3. 构建提示词
    const prompt = this.buildPrompt(template.promptTemplate, input);

    // 4. 选择AI模型
    const model = this.selectModel(options?.model || template.defaultModel);

    // 5. 调用AI服务
    const result = await this.callAI(model, prompt, {
      temperature: options?.temperature ?? 0.7,
      maxTokens: options?.maxTokens ?? 2000
    });

    // 6. 解析输出
    const output = this.parseOutput(result, template.outputSchema);

    // 7. 记录执行日志
    await this.logExecution(skillCode, input, output, result.usage);

    return output;
  }
}
```

### 6.4 AI调用封装

```typescript
// AI服务客户端
class AIClient {
  private providers: Map<string, AIProvider>;

  constructor() {
    this.providers = new Map([
      ['openai', new OpenAIProvider()],
      ['anthropic', new AnthropicProvider()],
      ['local', new LocalLLMProvider()]
    ]);
  }

  async generate(prompt: string, options: GenerateOptions): Promise<GenerateResult> {
    const provider = this.providers.get(options.provider || 'openai');
    if (!provider) {
      throw new Error(`Provider not found: ${options.provider}`);
    }

    const result = await provider.generate(prompt, {
      model: options.model,
      temperature: options.temperature ?? 0.7,
      maxTokens: options.maxTokens ?? 2000
    });

    return {
      text: result.text,
      usage: {
        promptTokens: result.usage.promptTokens,
        completionTokens: result.usage.completionTokens,
        totalTokens: result.usage.totalTokens
      },
      cost: this.calculateCost(options.provider, options.model, result.usage)
    };
  }

  private calculateCost(provider: string, model: string, usage: TokenUsage): number {
    // 根据不同提供商和模型计算成本
    const rates = {
      'openai': {
        'gpt-4': { input: 0.03, output: 0.06 },  // per 1K tokens
        'gpt-3.5-turbo': { input: 0.0015, output: 0.002 }
      },
      'anthropic': {
        'claude-3-opus': { input: 0.015, output: 0.075 },
        'claude-3-sonnet': { input: 0.003, output: 0.015 }
      }
    };

    const rate = rates[provider]?.[model];
    if (!rate) return 0;

    return (usage.promptTokens / 1000) * rate.input +
           (usage.completionTokens / 1000) * rate.output;
  }
}
```
