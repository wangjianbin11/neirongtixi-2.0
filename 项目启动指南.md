### 3.4 创建入口文件

```typescript
// src/index.ts
import express, { Application } from 'express';
import cors from 'cors';
import helmet from 'helmet';
import morgan from 'morgan';
import { config } from './config';
import { errorHandler } from './middleware/errorHandler';
import { apiRouter } from './api';
import { logger } from './utils/logger';

const app: Application = express();

// 中间件
app.use(helmet());
app.use(cors({ origin: config.cors.origin }));
app.use(express.json({ limit: '10mb' }));
app.use(express.urlencoded({ extended: true }));
app.use(morgan('combined', { stream: { write: (msg) => logger.info(msg.trim()) } }));

// 健康检查
app.get('/health', (req, res) => {
  res.json({ status: 'ok', timestamp: new Date().toISOString() });
});

// API路由
app.use(config.api.prefix, apiRouter);

// 错误处理
app.use(errorHandler);

// 启动服务器
const PORT = config.server.port;
app.listen(PORT, () => {
  logger.info(`🚀 Server running on port ${PORT}`);
  logger.info(`📝 Environment: ${config.server.env}`);
  logger.info(`🔗 API: http://localhost:${PORT}${config.api.prefix}`);
});

export default app;
```

### 5.3 创建FastAPI入口

```python
# app/main.py
from fastapi import FastAPI
from fastapi.middleware.cors import CORSMiddleware
from loguru import logger
from app.api import router
from app.core.config import settings

app = FastAPI(
    title="ASG Content AI Service",
    description="AI-powered content generation service",
    version="1.0.0"
)

# CORS配置
app.add_middleware(
    CORSMiddleware,
    allow_origins=settings.CORS_ORIGINS,
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)

# 路由
app.include_router(router, prefix="/api/v1")

# 健康检查
@app.get("/health")
async def health_check():
    return {"status": "ok", "service": "ai-service"}

# 启动事件
@app.on_event("startup")
async def startup_event():
    logger.info("🤖 AI Service starting up...")

@app.on_event("shutdown")
async def shutdown_event():
    logger.info("🤖 AI Service shutting down...")
```

## 六、数据库配置

### 6.1 Docker Compose快速启动

```bash
cd ..

# 创建 docker-compose.yml
cat > docker-compose.dev.yml << 'EOF'
version: '3.8'

services:
  postgres:
    image: postgres:15-alpine
    container_name: asg-postgres
    ports:
      - "5432:5432"
    environment:
      POSTGRES_USER: asg_user
      POSTGRES_PASSWORD: asg_password
      POSTGRES_DB: asg_content
    volumes:
      - postgres_data:/var/lib/postgresql/data
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U asg_user"]
      interval: 10s
      timeout: 5s
      retries: 5

  mongo:
    image: mongo:7
    container_name: asg-mongo
    ports:
      - "27017:27017"
    environment:
      MONGO_INITDB_ROOT_USERNAME: asg_user
      MONGO_INITDB_ROOT_PASSWORD: asg_password
      MONGO_INITDB_DATABASE: asg_content
    volumes:
      - mongo_data:/data/db
    healthcheck:
      test: ["CMD", "mongosh", "--eval", "db.adminCommand('ping')"]
      interval: 10s
      timeout: 5s
      retries: 5

  redis:
    image: redis:7-alpine
    container_name: asg-redis
    ports:
      - "6379:6379"
    volumes:
      - redis_data:/data
    healthcheck:
      test: ["CMD", "redis-cli", "ping"]
      interval: 10s
      timeout: 5s
      retries: 5

volumes:
  postgres_data:
  mongo_data:
  redis_data:
EOF

# 启动数据库服务
docker-compose -f docker-compose.dev.yml up -d
```

## 八、常见问题

### Q1: 端口被占用怎么办？

```bash
# Windows查看端口占用
netstat -ano | findstr :4000
# 结束进程
taskkill /PID <进程ID> /F

# Linux/Mac查看端口占用
lsof -i :4000
# 结束进程
kill -9 <进程ID>
```

### Q2: 数据库连接失败？

```bash
# 检查Docker容器状态
docker-compose -f docker-compose.dev.yml ps

# 查看日志
docker-compose -f docker-compose.dev.yml logs postgres

# 重启数据库
docker-compose -f docker-compose.dev.yml restart postgres
```

### Q3: 如何重置数据库？

```bash
# 删除所有数据并重新创建
docker-compose -f docker-compose.dev.yml down -v
docker-compose -f docker-compose.dev.yml up -d

# 重新运行迁移
cd backend
npx prisma migrate reset
```
